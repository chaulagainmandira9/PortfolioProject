Example Code: Ore Quality Calculator

public class OreAnalyzer {
    public static void main(String[] args) {
        // Sample mining data analysis
        double[] samples = {23.4, 25.1, 22.8, 24.5};
        double average = calculateAverage(samples);

        System.out.println("Average ore quality: "
            + average + "% mineral content");
        if(average > 24.0) {
            System.out.println("✅ High quality - prioritize extraction");
        }
    }

    public static double calculateAverage(double[] values) {
        double sum = 0.0;
        for(double val : values) {
            sum += val;
        }
        return sum / values.length;
    }
}
